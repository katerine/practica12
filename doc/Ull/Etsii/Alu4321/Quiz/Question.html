# File ull-etsii-alu4321-quiz/lib/ull-etsii-alu4321-quiz.rb, line 136
def to_s
    out = "# #{@title}".colorize(:light_blue) + "\n"
    i = 1
    answers.each do |a|
        out << " [#{i}] #{a}\n"
        i += 1
    end
    out
end
